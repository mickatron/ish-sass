    variables

    responsive-settings

    $responsive-settings: (
	base: (
		breakpoint: 0px,
	 	width: 100%,
		gutter: 5%,
		//grid-gutter: 3%, // TODO: not working or implemented
		base-size: map-get($base, font-size),
    	line-height: map-get($base, line-height),
    	ratio: map-get($scales, minorThird),
    	header-height: 56px
	),
	mobile: (
		breakpoint: $mobile-start,
		//width: 100%,
		gutter: 10px,
	    base-size: 16px,
	   	line-height: 1.3,
	    ratio: map-get($scales, minorThird),
    	header-height: 56px
	), 
	tablet: (
		breakpoint: $tablet-start,
		//width: 100%,
		gutter: 5%,
		base-size: 18px,
	    base-line-height: 1.7,
	    ratio: map-get($scales, fourth),
    	header-height: 100vh
	),
	desktop: (
		breakpoint: $desktop-start,
		width: 1440px,
		//gutter: 15%,    
    	header-height: 100vh,
    	ratio: map-get($scales, fourth),
	),
	desktopLarge: (
		breakpoint:$desktop-high-res-start,
		base-size: 20px,
		grid-gutter: 5%,
    	ratio: map-get($scales, fourth),
	),
);

    Description

    Default responsive $responsive-settings as used in responsive mixins such as font-size and responsive-gutters.
